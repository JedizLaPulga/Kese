package main

import (
    "github.com/JedizLaPulga/kese"
    "github.com/JedizLaPulga/kese/context"
)

func main() {
    app := kese.New()
    
    app.GET("/", func(c *context.Context) error {
        return c.JSON(200, map[string]string{
            "message": "Hello from Kese!",
        })
    })
    
    app.Run(":8080")
}
